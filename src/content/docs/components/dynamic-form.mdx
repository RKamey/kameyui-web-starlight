            }
          }
        ],
      },
      {

        name: 'email',

            }

## API Reference
Para configurar el formulario, se pueden utilizar las siguientes propiedades:

| Propiedad | Descripción | Tipo | Valor por defecto | Opcional |
| --- | --- | --- | --- | --- |
| mode | Modo del formulario | 'create' \| 'update' | 'vertical' | Sí |
| title | Título del formulario | string | - | Sí |
| description | Descripción en el header del formulario | string | - | Sí |
| icon | Icono en el header del formulario | ReactElement | - | Sí |
| layout | Layout del formulario | 'vertical' \| 'horizontal' | 'vertical' | Sí |
| cols | Número de columnas del formulario | number | 1 | Sí |
| fields | Campos del formulario | FormField[] | FormField[][] | - | No |
| submitButtonText | Texto del botón de submit | string | 'Enviar' | Sí |
| onSubmit | Función que se ejecuta al hacer submit | (data: unkwown) => void | - | Sí |
| apiConfig | Configuración de la API | ApiConfig | - | Sí |
| initialData | Datos iniciales del formulario | Record&lt;string&#44; unknown&gt; | - | Sí |
| customCols | Columnas personalizadas | boolean | - | Sí |
